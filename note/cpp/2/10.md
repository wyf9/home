### 5

输入一个成绩,打印相应的等级

?> 等级:

- >= 90: A
- 80 ~ 90: B
- 70 ~ 80: C
- 60 ~ 70: D
- < 60: E

<details>
<summary>点击展开</summary>

```cpp
int e1;
cin >> e1;
string e2;
if (e1 > 90) {
    e2 = "A";
} else if (e1 > 80) {
    e2 = "B";
} else if (e1 > 70) {
    e2 = "C";
} else if (e1 > 60) {
    e2 = "D";
} else {
    e2 = "E";
}
cout << "等级: " << e2 << endl;
```

### 完整示例

<details>
<summary>点击展开</summary>

```cpp
#include <iostream>
using namespace std;

int main() {
    cout << "---1---" << endl;
    int a1, a2;
    cin >> a1 >> a2;
    cout << ((a1 > a2) ? a1 : a2);

    cout << endl << "---2---" << endl;
    float b1, b2, bS;
    cin >> b1 >> b2;
    bS = b1 * b2;
    cout << "S = " << bS << endl;

    cout << "---3---" << endl;
    int c1, c2;
    cin >> c1 >> c2;
    if (c1 % c2 == 0) {
        cout << "能整除" << endl;
    } else {
        cout << "不能整除" << endl;
    }

    cout << "---4---" << endl;
    int d1;
    cin >> d1;
    if (d1 % 2 == 0) {
        cout << "偶数" << endl;
    } else {
        cout << "奇数" << endl;
    }

    cout << "---5---" << endl;
    int e1;
    cin >> e1;
    string e2;
    if (e1 > 90) {
        e2 = "A";
    } else if (e1 > 80) {
        e2 = "B";
    } else if (e1 > 70) {
        e2 = "C";
    } else if (e1 > 60) {
        e2 = "D";
    } else {
        e2 = "E";
    }
    cout << "等级: " << e2 << endl;
    return 0;
}
```
